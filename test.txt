from typing import List, Tuple

def generate_month_windows(lag_months: int) -> List[Tuple[int, int]]:
    """
    Generate month windows for a given lag.

    Returns:
    - Rolling 3-month windows up to lag_months.
    - Cumulative windows: (1, 6), (7, 12), and (1, lag_months) if applicable.

    Args:
        lag_months (int): Number of months to look back (>=1)

    Raises:
        ValueError: If lag_months < 1
    """
    if lag_months < 1:
        raise ValueError("lag_months must be >= 1")

    windows = []

    for i in range(1, lag_months + 1, 3):
        start = i
        end = min(i + 2, lag_months)
        windows.append((start, end))

    if lag_months >= 6 and (1, 6) not in windows:
        windows.append((1, 6))
    if lag_months >= 12 and (7, 12) not in windows:
        windows.append((7, 12))
    if (1, lag_months) not in windows:
        windows.append((1, lag_months))

    return windows
