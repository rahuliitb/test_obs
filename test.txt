from pyspark.sql import functions as F

df_merged = df_merged.select(
    *[
        (
            F.floor(F.col(c) * 100) / 100
        ).alias(c)
        if c not in [
            cfg.application_id,
            "difference_in_months",
            cfg.main_borrower_flag,
            "month",
            "m_num_days_negative_balance"
        ]
        else F.col(c)
        for c in df_merged.columns
    ]
)
