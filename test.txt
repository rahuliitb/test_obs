def map_lightgbm_native_to_sklearn(params_cv):
    mapped = params_cv.copy()
    if "min_data_in_leaf" in params_cv:
        mapped["min_child_samples"] = params_cv["min_data_in_leaf"]
    if "min_sum_hessian_in_leaf" in params_cv:
        mapped["min_child_weight"] = params_cv["min_sum_hessian_in_leaf"]
    if "bagging_fraction" in params_cv:
        mapped["subsample"] = params_cv["bagging_fraction"]
    if "feature_fraction" in params_cv:
        mapped["colsample_bytree"] = params_cv["feature_fraction"]
    if "num_boost_round" in params_cv:
        mapped["n_estimators"] = params_cv["num_boost_round"]
    return mapped
